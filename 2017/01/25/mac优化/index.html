初始化系统

第三方开启

sudo spctl --master-disable

增加空白占位符

defaults write com.apple.dock persistent-apps -array-add '{"tile-type"="spacer-tile";}'; Killall Dock

隐藏应用使用透明图标显示

defaults write com.apple.dock showhidden -bool true; Killall Dock

恢复默认

defaults delete com.apple.Dock showhidden; Killall Dock

安装xcode插件

xcode-select --install

Anaconda brew报错

alias brew="env PATH=${PATH//$(pyenv root)\/shims:/} brew

conda create -n python2 python=2.7.13

git设置默认

git config --global user.email ""
git config --global user.name ""

homebrew配置

安装homebrew

/usr/bin/ruby -e "$(curl -fsSL https://raw.githubusercontent.com/Homebrew/install/master/install)"

export HOMEBREW_BOTTLE_DOMAIN=https://mirrors.tuna.tsinghua.edu.cn/homebrew-bottles

brew install autojump autoconf autocode automake curl xz tmux colormake zsh-autosuggestions zsh-syntax-highlighting
